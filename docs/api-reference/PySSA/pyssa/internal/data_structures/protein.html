206    def add_protein_structure_data_from_pdb_db(self, a_pdb_id) -> None:
207        """Adds protein structure data based on a protein from the PDB database."""
208        tmp_pdb_filepath = pathlib.Path(f"{constants.CACHE_PROTEIN_DIR}/{a_pdb_id}.pdb")
209        bio_data.download_pdb_file(a_pdb_id, tmp_pdb_filepath)
210        self.chains = protein_operations.get_protein_chains(
211            self._pymol_molecule_object,
212            tmp_pdb_filepath.parent,
213            tmp_pdb_filepath.name,
214        )
215        self._pdb_data = bio_data.parse_pdb_file(tmp_pdb_filepath)
216        self.check_states_and_reduce_to_one_state_if_necessary()
217        try:
218            os.remove(str(pathlib.Path(f"{constants.CACHE_PROTEIN_DIR}/{a_pdb_id}.pdb")))
219        except Exception as e:
220            logger.error(f"Could not delete pdb file! Ran into error: {e}")

326    def write_fasta_file(self, a_filepath: pathlib.Path) -> None:
327        """Writes a fasta file the specified filepath.
328
329        Raises:
330            IllegalArgumentError: If the file is None or the file could not be found.
331            DirectoryDoesNotExistError: If the parent directory of the filepath could not be found.
332        """
333        # <editor-fold desc="Checks">
334        if a_filepath is None or not os.path.exists(a_filepath):
335            exception.IllegalArgumentError("")
336        if not os.path.exists(a_filepath.parent):
337            raise exception.DirectoryDoesNotExistError("")
338
339        # </editor-fold>
340
341        try:
342            fasta_file: TextIO = open(f"{a_filepath}/{self._pymol_molecule_object}.fasta", "w")
343        except OSError:
344            raise exception.FastaFilesNotCreatedError("")
345
346        # Writes fasta file header
347        fasta_file.write(f">{self._pymol_molecule_object}\n")
348
349        # Writes fasta file sequence content
350        i: int = 0
351        seq_objs: list[sequence.Sequence] = self.get_protein_sequences()
352        for tmp_sequence in seq_objs:
353            if i == len(self.get_protein_sequences()) - 1:
354                # should be the last entry
355                fasta_file.write(tmp_sequence.sequence)
356            else:
357                fasta_file.write(f"{tmp_sequence.sequence}:")
358            i += 1
359
360        logger.info(f"Fasta file for sequence {self._pymol_molecule_object} written.")
361        fasta_file.close()

391    def load_protein_in_pymol(self) -> None:
392        """Load a protein in PyMOL.
393
394        Raises:
395            UnableToCreatePdbFileError:If the pdb file cannot be created.
396            UnableToOpenPdbFileError:If the pdb file cannot be opened.
397            UnableToLoadPdbFileError:If the pdb file cannot be loaded.
398        """
399        # <editor-fold desc="Checks">
400        pdb_filepath = pathlib.Path(f"{constants.CACHE_PROTEIN_DIR}/{self._pymol_molecule_object}.pdb")
401        if not os.path.exists(constants.CACHE_PROTEIN_DIR):
402            os.mkdir(constants.CACHE_PROTEIN_DIR)
403
404        # </editor-fold>
405
406        if len(self._pdb_data) == 0:
407            raise ValueError("No pdb data in current object!")
408
409        try:
410            bio_data.build_pdb_file(self._pdb_data, pathlib.Path(f"{constants.CACHE_PROTEIN_DIR}/{self._pymol_molecule_object}.pdb"))
411        except exception.IllegalArgumentError:
412            logger.error(f"The argument pdb data is not usable: {self._pdb_data}.")
413            raise exception.UnableToCreatePdbFileError("")
414        except exception.DirectoryNotFoundError:
415            logger.error(f"The argument pdb_filepath is illegal: {pdb_filepath}!")
416            raise exception.UnableToCreatePdbFileError("")
417        except PermissionError:
418            logger.error(f"The argument pdb_filepath is illegal: {pdb_filepath}!")
419            raise exception.UnableToCreatePdbFileError("")
420        except exception.UnableToOpenFileError:
421            logger.error("pdb file could not be opened for writing.")
422            raise exception.UnableToOpenFileError("")
423
424        try:
425            pymol_io.load_protein(constants.CACHE_PROTEIN_DIR, f"{self._pymol_molecule_object}.pdb", self._pymol_molecule_object)
426        except exception.UnableToLoadProteinError:
427            logger.error("Protein can not be loaded in PyMOL!")
428            raise exception.UnableToLoadProteinError

493    def clean_protein(self, new_protein: bool = False):  # noqa: ANN201 #TODO: needs to be redone
494        """Cleans the protein from all sugar and solvent molecules.
495
496        Args:

503                protein_operations.remove_organic_molecules_in_protein()
504            except pymol.CmdException:
505                return  # noqa: RET502 #TODO: needs more thoughts
506            # tmp_full_pdb_path = pathlib.Path(f"{constants.CACHE_PROTEIN_DIR}/{self._id}.pdb")
