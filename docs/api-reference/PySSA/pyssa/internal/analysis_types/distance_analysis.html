193    def do_analysis_in_pymol(self) -> None:
194        """This function does the distance analysis of the protein pair.
195
196        Raises:
197            UnableToDoAnalysisError: If the analysis fails in PyMOL.
198        """
199        logger.info("Start of do_analysis_in_pymol() method.")
200        try:
201            self._protein_pair_for_analysis.load_protein_pair_in_pymol()
202            logger.info(f"Loaded protein pair: " f"{self._protein_pair_for_analysis.name} in pymol session.")
203            self._protein_pair_for_analysis.color_protein_pair()
204            logger.info(f"Colored protein pair: " f"{self._protein_pair_for_analysis.name} in pymol session.")
205
206            logger.debug(
207                f"Protein 1 selection string: "
208                f"{self._protein_pair_for_analysis.protein_1.pymol_selection.selection_string} "
209                f"{self._protein_pair_for_analysis.protein_1.pymol_selection}",
210            )
211            logger.debug(
212                f"Protein 2 selection string: "
213                f"{self._protein_pair_for_analysis.protein_2.pymol_selection.selection_string} "
214                f"{self._protein_pair_for_analysis.protein_2.pymol_selection}",
215            )
216
217            align_results = self.align_protein_pair_for_analysis()
218            logger.info(f"Aligned protein pair: " f"{self._protein_pair_for_analysis.name} in pymol session.")
219
220            fasta_prot_1 = cmd.get_fastastr(self._protein_pair_for_analysis.protein_1.pymol_selection.selection_string)
221            logger.debug(fasta_prot_1)
222            logger.debug(fasta_prot_1[fasta_prot_1.find("\n") :])
223            seq_length_prot_1 = len(fasta_prot_1[fasta_prot_1.find("\n") :])
224            logger.debug(seq_length_prot_1)
225            # Total no. of residues
226            self.rmsd_dict = {
227                "rmsd": str(round(align_results[0], 2)),
228                "aligned_residues": f"{str(align_results[1])} / {seq_length_prot_1}",
229            }
230
231            distances = protein_pair_util.calculate_distance_between_ca_atoms(
232                self._protein_pair_for_analysis.protein_1.get_molecule_object(),
233                self._protein_pair_for_analysis.protein_2.get_molecule_object(),
234            )
235            logger.info(f"Calculated distances of protein pair: " f"{self._protein_pair_for_analysis.name}.")
236            self.distance_analysis_data = distances
237            self.analysis_results = results.DistanceAnalysisResults(
238                self.distance_analysis_data,
239                pymol_io.convert_pymol_session_to_base64_string(self._protein_pair_for_analysis.name),
240                self.rmsd_dict["rmsd"],
241                self.rmsd_dict["aligned_residues"],
242            )
243        except exception.UnableToDoAnalysisError:
244            logger.error("The analysis in PyMOL failed!")
245            raise exception.UnableToDoAnalysisError

247    def take_image_of_protein_pair(
248        self,
249        representation: str,
250        filename: str,
251        selection: str = "",
252        ray_shadows: bool = False,
253        opaque_background: int = 0,
254        take_images: bool = True,
255    ) -> None:
256        """This function takes an image of the whole Protein/Protein pair.
257
258        Note:
259            The png file will be saved under the relative path
260            (if export_data_dir = "data/results"):
261            ``data/results/images``
262
263        Args:
264            representation (str): defines the type of molecular representation like cartoon or ribbon.
265            filename (str): name of the png image file.
266            selection (str): the atoms which MUST NOT displayed in the image.
267            ray_shadows (bool): false if no shadows, true if shadows should be displayed.
268            opaque_background (int, optional): 0 for a transparent background and 1 for a white background.
269            take_images: Is a boolean, indicating to take images or not.
270
271        Raises:
272            ValueError: If opaque_background is not 0 or 1.
273            UnableToColorProteinPairError: If the protein pair is not able to colored.
274        """
275        # determine the option for ray_shadows
276        if not ray_shadows:
277            opt_ray_shadows: str = "off"
278        else:
279            opt_ray_shadows: str = "on"
280
281        representation = "cartoon"
282        cmd.show(representation)
283
284        if selection != "":
285            cmd.hide(representation, selection)
286
287        aln_obj_representation: str = "cgo"
288        cmd.hide(aln_obj_representation, "all")
289        cmd.orient()
290        cmd.center()
291        # set image parameters
292        cmd.bg_color(constants.PYMOL_DEFAULT_BACKGROUND_COLOR)
293        cmd.set("ray_trace_mode", constants.PYMOL_DEFAULT_RAY_TRACE_MODE)
294        cmd.set("antialias", constants.PYMOL_DEFAULT_ANTIALIAS)
295        cmd.set("ray_shadows", opt_ray_shadows)
296        cmd.set("ray_opaque_background", opaque_background)
297        graphic_operations.setup_default_session_graphic_settings()
298
299        try:
300            self._protein_pair_for_analysis.color_protein_pair()
301        except exception.UnableToColorProteinPairError:
302            logger.error("Unable color protein pair.")
303            raise exception.UnableToColorProteinPairError("")
304
305        cmd.scene(
306            key=f"{self._protein_pair_for_analysis.protein_1.get_molecule_object()}-"
307            f"{self._protein_pair_for_analysis.protein_2.get_molecule_object()}",
308            action="store",
309        )
310
311        if take_images is True:
312            # check if path exists where the data will be exported,
313            # if not the directory will be created
314            if not os.path.exists(f"{constants.SCRATCH_DIR_STRUCTURE_ALN_IMAGES_DIR}"):
315                os.mkdir(f"{constants.SCRATCH_DIR_STRUCTURE_ALN_IMAGES_DIR}")
316            # save image as 300 dpi png image
317            cmd.ray(2400, 2400, renderer=0)
318            cmd.png(f"{constants.SCRATCH_DIR_STRUCTURE_ALN_IMAGES_DIR}/{filename}.png", dpi=300)

320    def take_image_of_interesting_regions(
321        self,
322        cutoff: float,
323        filename: str,
324        ray_shadows: bool = False,
325        opaque_background: int = 0,
326        take_images: bool = True,
327    ) -> None:
328        """This function takes images of interesting regions of the alignment.
329
330        Args:
331            cutoff (float): Defines a border of which the specific regions begin, if the distance is greater than
332                            the cutoff, the amino acid is categorized as "interesting".
333            filename (str): Is the name of the png image file.
334            ray_shadows (bool): Is false if no shadows, true if shadows should be displayed.
335            opaque_background (int): 0 for a transparent background and 1 for a white background.
336            take_images (bool): Flag if images should be made or not.
337
338        Raises:
339            UnableToColorProteinPairError: If the protein pair is not able to colored.
340        """
341        # set default parameters
342        cmd.set("label_size", 14)
343        cmd.set("label_font_id", 13)
344        cmd.set("label_color", "hotpink")
345        cmd.set("depth_cue", 0)
346        # cmd.set("fog_start", 0.6)
347        cmd.bg_color(constants.PYMOL_DEFAULT_BACKGROUND_COLOR)
348
349        j: int = 0
350        representation: str = "ribbon"
351
352        cmd.hide("cartoon", "all")
353        cmd.show(representation, "all")
354        graphic_operations.setup_default_session_graphic_settings()
355
356        try:
357            self._protein_pair_for_analysis.color_protein_pair()
358        except exception.UnableToColorProteinPairError:
359            logger.error("Unable color protein pair.")
360            raise exception.UnableToColorProteinPairError("")
361
362        i: int = 0
363        for distance_value in self.analysis_results.distance_data.get("distance"):
364            if float(distance_value) > float(cutoff):
365                # here algorithm for image
366                ref_pos_array: np.ndarray = self.analysis_results.distance_data.get("ref_pos")
367                ref_pos: int = ref_pos_array[i]
368                ref_chain_array: np.ndarray = self.analysis_results.distance_data.get("ref_chain")
369                ref_chain: str = ref_chain_array[i]
370                model_pos_array: np.ndarray = self.analysis_results.distance_data.get("model_pos")
371                model_pos: int = model_pos_array[i]
372                model_chain_array: np.ndarray = self.analysis_results.distance_data.get(
373                    "model_chain",
374                )  # pylint: disable=line-too-long
375                model_chain: str = model_chain_array[i]
376
377                measurement_obj: str = f"measure{j}"
378                # create two atoms for the get_distance command
379                atom1: str = (
380                    f"/{self._protein_pair_for_analysis.protein_1.get_molecule_object()}//" f"{ref_chain}/{ref_pos}/CA"
381                )
382                atom2: str = (
383                    f"/{self._protein_pair_for_analysis.protein_2.get_molecule_object()}//"
384                    f"{model_chain}/{model_pos}/CA"
385                )
386                # zoom to reference amino acid
387                cmd.zoom(
388                    f"/{self._protein_pair_for_analysis.protein_1.get_molecule_object()}//" f"{ref_chain}/{ref_pos}",
389                    10,
390                )
391                # create distance object with get_distance command
392                cmd.distance(measurement_obj, atom1, atom2)
393                cmd.label(atom1, "'%s-%s' % (resn, resi)")
394                cmd.label(atom2, "'%s-%s' % (resn, resi)")
395                cmd.set("label_position", (0, 0, 10))
396                # determine the option for ray_shadows
397                if ray_shadows is False:
398                    opt_ray_shadows: str = "off"
399                else:
400                    opt_ray_shadows: str = "on"
401                # set image parameters
402                cmd.bg_color(constants.PYMOL_DEFAULT_BACKGROUND_COLOR)
403                cmd.set("ray_trace_mode", 3)
404                cmd.set("antialias", 2)
405                cmd.set("ray_shadows", opt_ray_shadows)
406                cmd.set("ray_opaque_background", opaque_background)
407
408                cmd.scene(key=f"{ref_pos}-{model_pos}", action="store")
409
410                if take_images is True:
411                    # check if path exists where the data will be exported,
412                    # if not the directory will be created
413                    if not os.path.exists(constants.SCRATCH_DIR_STRUCTURE_ALN_IMAGES_INTERESTING_REGIONS_DIR):
414                        os.mkdir(constants.SCRATCH_DIR_STRUCTURE_ALN_IMAGES_INTERESTING_REGIONS_DIR)
415                    # save image as 300 dpi png image
416                    cmd.ray(2400, 2400, renderer=0)
417                    tmp_filename: str = f"{filename}_{ref_pos}.png"
418                    cmd.png(
419                        f"{constants.SCRATCH_DIR_STRUCTURE_ALN_IMAGES_INTERESTING_REGIONS_DIR}/{tmp_filename}",
420                        dpi=300,
421                    )
422                # hide created labels
423                cmd.hide("labels", atom1)
424                cmd.hide("labels", atom2)
425                cmd.hide("labels", measurement_obj)
426                cmd.hide("dashes", measurement_obj)
427                i += 1
428                j += 1
429            else:
430                i += 1

432    def serialize_distance_analysis(
433        self,
434        xml_distance_analysis_element,  # noqa: ANN001 #TODO: needs to be checked
435    ) -> None:
436        """This function serialize the protein pair object."""
437        tmp_distance_analysis = ElementTree.SubElement(
438            xml_distance_analysis_element,
439            element_names.DISTANCE_ANALYSIS,
440        )
441        tmp_distance_analysis.set(attribute_names.DISTANCE_ANALYSIS_NAME, str(self.name))
442        tmp_distance_analysis.set(attribute_names.DISTANCE_ANALYSIS_CUTOFF, str(self.cutoff))
443        tmp_distance_analysis.set(attribute_names.DISTANCE_ANALYSIS_CYCLES, str(self.cycles))
444
445        logger.debug(f"Serialization of: {self.name}")
446        logger.debug(self.analysis_results)
447
448        self.analysis_results.serialize_distance_analysis_results(tmp_distance_analysis)
449        tmp_session_data = ElementTree.SubElement(
450            tmp_distance_analysis,
451            element_names.DISTANCE_ANALYSIS_SESSION,
452        )
453        tmp_session_data.set(
454            attribute_names.PROTEIN_PAIR_SESSION,
455            pymol_io.convert_pymol_session_to_base64_string(self.name),
456        )
