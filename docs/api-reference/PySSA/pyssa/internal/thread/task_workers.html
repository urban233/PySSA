 54def setup_worker_for_work(tmp_thread: QtCore.QThread, tmp_worker, return_value_func) -> QtCore.QThread:  # noqa: ANN001
 55    """Sets up the worker for the given thread."""
 56    tmp_worker.moveToThread(tmp_thread)
 57    tmp_thread.started.connect(tmp_worker.run)
 58    tmp_worker.finished.connect(tmp_thread.quit)
 59    tmp_worker.finished.connect(tmp_worker.deleteLater)
 60    tmp_thread.finished.connect(tmp_thread.deleteLater)
 61    tmp_worker.return_value.connect(return_value_func)
 62    return tmp_thread

119        """Creates a numpy array of QListWidgetItem objects.

156class CreateUseProjectWorker(QObject):
157    """Class for a worker which creates the project on the use page."""
158

160    progress = pyqtSignal(int)
161    return_value = pyqtSignal(list)
162    workspace_path: str
163    proteins_to_copy: list
164
165    def __init__(self, workspace, proteins_to_copy) -> None:  # noqa: ANN001 #TODO: needs to be checked
166        """Constructor.
167
168        Args:
169            workspace: the path to the workspace.
170            proteins_to_copy: the proteins for the new project.
171        """
172        super().__init__()
173        self.workspace_path = workspace
174        self.proteins_to_copy = proteins_to_copy
175
176    def run(self) -> None:
177        """This function is a reimplementation of the QObject run method."""
178        proteins_for_new_project = []
179        protein_infos = (tools.scan_workspace_for_non_duplicate_proteins(pathlib.Path(self.workspace_path)))[0]
180        for tmp_protein in self.proteins_to_copy:
181            for tmp_protein_info in protein_infos:
182                if tmp_protein_info.name == tmp_protein:
183                    """Var: project_proteins is a list which contains all proteins from a single project"""
184                    xml_deserializer = filesystem_io.XmlDeserializer(
185                        pathlib.Path(f"{self.workspace_path}/{tmp_protein_info.project_name}.xml"),
186                    )
187                    for xml_protein in xml_deserializer.xml_root.iter(element_names.PROTEIN):
188                        if xml_protein.attrib[attribute_names.ID] == tmp_protein_info.id:
189                            basic_information = xml_protein.attrib
190                            pdb_lines = []
191                            session_data_base64 = ""
192                            for tmp_data in xml_protein:
193                                if tmp_data.tag == "pdb_data":
194                                    for tmp_atom in tmp_data.findall("atom"):
195                                        pdb_lines.append(tmp_atom.text)
196                                elif tmp_data.tag == "session_data":
197                                    session_data_base64 = tmp_data.attrib[attribute_names.PROTEIN_SESSION]
198                                else:
199                                    raise ValueError
200                            tmp_protein_obj = protein.Protein(
201                                molecule_object=basic_information[attribute_names.PROTEIN_MOLECULE_OBJECT],
202                                pdb_xml_string=xml_protein,
203                            )
204                            tmp_protein_obj.set_all_attributes(basic_information, pdb_lines, session_data_base64)
205            proteins_for_new_project.append(tmp_protein_obj)
206
207        self.return_value.emit(proteins_for_new_project)
208        self.finished.emit()

222        self,
223        protein_pair_of_results,  # noqa: ANN001 #TODO: needs to be checked
224        app_project_xml_filepath,  # noqa: ANN001 #TODO: needs to be checked
225    ) -> None:
226        """Constructor.
227
228        Args:
229            protein_pair_of_results: the protein pair which results should be loaded.
230            app_project_xml_filepath: the filepath of the project xml file.
231        """
232        super().__init__()
233        self.protein_pair_of_results = protein_pair_of_results
234        self.image_type = constants.IMAGES_NONE
235        self.app_project_xml_filepath = app_project_xml_filepath

237    def run(self) -> None:
238        """This function is a reimplementation of the QObject run method."""
239        shutil.rmtree(constants.CACHE_DIR)
240        os.mkdir(constants.CACHE_DIR)
241        os.mkdir(constants.CACHE_IMAGES)
242
243        filesystem_io.XmlDeserializer(pathlib.Path(self.app_project_xml_filepath)).deserialize_analysis_images(
244            self.protein_pair_of_results.name,
245            self.protein_pair_of_results.distance_analysis.analysis_results,
246        )
247        if (
248            len(self.protein_pair_of_results.distance_analysis.analysis_results.structure_aln_image) != 0
249            and len(self.protein_pair_of_results.distance_analysis.analysis_results.interesting_regions_images) != 0
250        ):
251            # if both image types were made during analysis
252            self.protein_pair_of_results.distance_analysis.analysis_results.create_image_png_files_from_base64()
253            self.image_type = constants.IMAGES_ALL
254        elif (
255            len(self.protein_pair_of_results.distance_analysis.analysis_results.structure_aln_image) != 0
256            and len(self.protein_pair_of_results.distance_analysis.analysis_results.interesting_regions_images) == 0
257        ):
258            # only struct align image were made
259            self.protein_pair_of_results.distance_analysis.analysis_results.create_image_png_files_from_base64()
260            self.image_type = constants.IMAGES_STRUCT_ALN_ONLY
261        else:
262            # no images were made
263            self.image_type = constants.IMAGES_NONE
264
265        self.return_value.emit(self.image_type)
266        self.finished.emit()

380                )

382                interesting_region_filepaths,
383            )
384            shutil.rmtree(constants.SCRATCH_DIR_IMAGES)

417    """This class is a worker class for the prediction process.
418

446    def __init__(
447        self,
448        table_prot_to_predict: QtWidgets.QTableWidget,
449        prediction_config: prediction_configuration.PredictionConfiguration,
450        app_project: "project.Project",
451    ) -> None:
452        """Constructor.
453
454        Args:
455            table_prot_to_predict:
456                pyqt table which contains the proteins to predict
457            prediction_config:
458                the prediction_config object
459            app_project:
460                current project
461

463            ValueError: raised if an argument is illegal
464        """
465        super().__init__()
466        # <editor-fold desc="Checks">

468        safeguard.Safeguard.check_if_value_is_not_none(prediction_config, logger)
469        safeguard.Safeguard.check_if_value_is_not_none(app_project, logger)
470
471        # </editor-fold>
472
473        self.table = table_prot_to_predict
474        self.prediction_configuration = prediction_config
475        self.app_project = app_project
476
477    def run(self) -> None:
478        """This function is a reimplementation of the QObject run method. It does the structure prediction."""
479        predictions: list[
480            prediction_protein_info.PredictionProteinInfo
481        ] = prediction_util.get_prediction_name_and_seq_from_table(self.table)
482        structure_prediction_obj = structure_prediction.StructurePrediction(
483            predictions,
484            self.prediction_configuration,
485            self.app_project,
486        )
487        structure_prediction_obj.create_tmp_directories()
488        logger.info("Tmp directories were created.")
489
490        # Create fasta files for prediction
491        try:
492            structure_prediction_obj.create_fasta_files_for_prediction()
493        except exception.FastaFilesNotCreatedError:
494            logger.error("Fasta files were not created.")
495            self.finished.emit(exit_codes.ERROR_WRITING_FASTA_FILES[0], exit_codes.ERROR_WRITING_FASTA_FILES[1])
496            return
497        except exception.FastaFilesNotFoundError:
498            logger.error("Fasta files were not found.")
499            self.finished.emit(exit_codes.ERROR_FASTA_FILES_NOT_FOUND[0], exit_codes.ERROR_FASTA_FILES_NOT_FOUND[1])
500            return
501        except Exception as e:
502            logger.error(f"Unexpected error: {e}")
503            self.finished.emit(exit_codes.EXIT_CODE_ONE_UNKNOWN_ERROR[0], exit_codes.EXIT_CODE_ONE_UNKNOWN_ERROR[1])
504            return
505        else:
506            logger.info("Fasta files were successfully created.")
507
508        # Run structure prediction
509        try:
510            structure_prediction_obj.run_prediction()
511        except exception.PredictionEndedWithError:
512            logger.error("Prediction ended with error.")
513            self.finished.emit(exit_codes.ERROR_PREDICTION_FAILED[0], exit_codes.ERROR_PREDICTION_FAILED[1])
514            return
515        else:
516            logger.info("Prediction process finished.")
517
518        try:
519            structure_prediction_obj.move_best_prediction_models()
520            logger.info("Saved predicted pdb file into XML file.")
521        except exception.UnableToFindColabfoldModelError:
522            logger.error("Could not move rank 1 model, because it does not exists.")
523            self.finished.emit(
524                exit_codes.ERROR_COLABFOLD_MODEL_NOT_FOUND[0],
525                exit_codes.ERROR_COLABFOLD_MODEL_NOT_FOUND[1],
526            )
527            return
528        except FileNotFoundError:
529            logger.error("Could not move rank 1 model, because it does not exists.")
530            self.finished.emit(
531                exit_codes.ERROR_COLABFOLD_MODEL_NOT_FOUND[0],
532                exit_codes.ERROR_COLABFOLD_MODEL_NOT_FOUND[1],
533            )
534            return
535        except Exception as e:
536            logger.error(f"Unexpected error: {e}")
537            logger.error("Could not move rank 1 model, because it does not exists.")
538            self.finished.emit(exit_codes.EXIT_CODE_ONE_UNKNOWN_ERROR[0], exit_codes.EXIT_CODE_ONE_UNKNOWN_ERROR[1])
539            return
540        subprocess.run(["wsl", "--shutdown"])
541        logger.info("WSL gets shutdown.")
542        self.finished.emit(exit_codes.EXIT_CODE_ZERO[0], exit_codes.EXIT_CODE_ZERO[1])
