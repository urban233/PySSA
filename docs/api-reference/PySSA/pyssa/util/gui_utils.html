 71def choose_directory(self, txt_box_dir: QtWidgets.QLineEdit) -> None:  # noqa: ANN001
 72    """This function is for choosing a filepath with the QFileDialog.
 73
 74    Example:
 75        txt_box_dir: self.ui.txt_workspace_dir
 76
 77    Args:
 78        self:
 79            main_window object
 80        txt_box_dir:
 81            qt textbox object which holds the current path
 82    """
 83    current_file_path = pathlib.Path(txt_box_dir.text())
 84    new_file_path = pathlib.Path(
 85        QtWidgets.QFileDialog.getExistingDirectory(
 86            self,
 87            "Open Directory",
 88            str(current_file_path),
 89            options=QtWidgets.QFileDialog.ShowDirsOnly,
 90        ),
 91    )
 92    rvoid(os.access(new_file_path, os.W_OK))
 93    if new_file_path != pathlib.Path(".") and os.access(new_file_path, os.W_OK):
 94        txt_box_dir.setText(str(new_file_path))
 95    elif new_file_path != pathlib.Path(".") and not os.access(new_file_path, os.W_OK):
 96        basic_boxes.ok(
 97            "Permission error",
 98            "You do not have write permissions for this directroy. Please choose another one.",
 99            QMessageBox.Warning,
100        )
101        txt_box_dir.setText(str(current_file_path))
102    else:
103        txt_box_dir.setText(str(current_file_path))

134def error_dialog_settings(message: str, message_detail: str, settings_obj: "settings.Settings") -> None:
135    """This function creates an error dialog, which can be customized.
136
137    Args:
138        message:
139            text which should get displayed in the dialog
140        message_detail:
141            additional information
142        settings_obj:
143            settings object of the MainWindow class
144    """
145    msg = QMessageBox()

155    msg.exec_()
156
157    # button logic
158    if msg.clickedButton() == restore_settings_button:
159        tools.restore_default_settings(settings_obj)
