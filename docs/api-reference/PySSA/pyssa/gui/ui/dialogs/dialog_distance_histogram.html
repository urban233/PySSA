154    def plot_histogram(self) -> None:
155        """Plots the histogram."""
156        # Clear any existing plot
157        self.plot_widget.figure.clear()
158
159        distance_data: dict[
160            str,
161            np.ndarray,
162        ] = self.protein_pair_for_analysis.distance_analysis.analysis_results.distance_data
163        distance_list = copy.deepcopy(distance_data[pyssa_keys.ARRAY_DISTANCE_DISTANCES])
164        distance_list.sort()
165        length = len(distance_list)
166        max_distance = distance_list[length - 1]
167
168        # Create an axis and plot a histogram
169        ax = self.plot_widget.figure.add_subplot(111)
170        n, bins, patches = ax.hist(
171            distance_list,
172            bins=np.arange(0, max_distance + 0.25, 0.25),
173            orientation="horizontal",
174            rwidth=0.7,
175            color="#4B91F7",
176        )
177
178        # Add labels to the non-zero frequency histogram bars
179        for bin_value, patch in zip(n, patches):
180            x = patch.get_x() + patch.get_width() + 0.1
181            y = patch.get_y() + patch.get_height() / 2
182            ax.annotate(
183                f"{bin_value}",
184                xy=(x, y),
185                xycoords="data",
186                xytext=(3, 0),
187                textcoords="offset points",
188                ha="left",
189                va="center",
190            )
191
192        # Move the entire x-axis to the top
193        ax.xaxis.tick_top()
194        ax.xaxis.set_label_position("top")
195
196        # Calculate the midpoints between bin edges
197        bin_midpoints = (bins[:-1] + bins[1:]) / 2
198        # Set y-ticks at the bin midpoints
199        ax.set_yticks(bin_midpoints)
200        # Set custom tick labels
201        custom_labels = [f"{bin_start} - {bin_end}" for bin_start, bin_end in zip(bins[:-1], bins[1:])]
202        ax.set_yticklabels(custom_labels)
203        # Set axis labels
204        ax.set_xlabel("Count")
205        ax.set_ylabel("Bins")
206        # Invert the y-axis
207        ax.invert_yaxis()
208        # Remove the spines where no axis are present
209        ax.spines["right"].set_visible(False)
210        ax.spines["bottom"].set_visible(False)
211
212        # Remove the spines where no axis are present
213        ax.spines["right"].set_visible(False)
214        ax.spines["bottom"].set_visible(False)
215
216        # Set axis labels
217        ax.set_xlabel("Frequency of C-α distances")
218        ax.set_ylabel("Distance in Å")
219
220        # Adjust subplot parameters to reduce white space
221        self.plot_widget.figure.tight_layout()
222
223        # Refresh the canvas
224        self.plot_widget.canvas.draw()
225
226    def scroll_down(self) -> None:
227        """Scroll down the histogram."""
228        if self.scroll_pos == "up":
229            self.histogram_pos = (self.histogram_pos[0] + 0.3, self.histogram_pos[1] + 0.3)
230        self.view_box.setRange(yRange=[self.histogram_pos[0], self.histogram_pos[1]])
231        self.histogram_pos = (self.histogram_pos[0] + 0.3, self.histogram_pos[1] + 0.3)
232        self.scroll_pos = "down"
233
234    def scroll_up(self) -> None:
235        """Scroll up the histogram."""
236        if self.scroll_pos == "down":
237            self.histogram_pos = (self.histogram_pos[0] - 0.6, self.histogram_pos[1] - 0.6)
238        else:
239            self.histogram_pos = (self.histogram_pos[0] - 0.3, self.histogram_pos[1] - 0.3)
240        self.view_box.setRange(yRange=[self.histogram_pos[0], self.histogram_pos[1]])
241        self.scroll_pos = "up"
