440    def create_distance_histogram(self):
441        distance_data: dict[
442            str,
443            np.ndarray,
444        ] = self.protein_pair_for_analysis.distance_analysis.analysis_results.distance_data
445        distance_list = copy.deepcopy(distance_data[pyssa_keys.ARRAY_DISTANCE_DISTANCES])
446        distance_list.sort()
447        length = len(distance_list)
448        max_distance = distance_list[length - 1]
449
450        n, self.bins, self.patches = self._ax_hist.hist(
451            distance_list,
452            bins=np.arange(0, max_distance + 0.25, 0.25),
453            orientation="horizontal",
454            rwidth=0.7,
455            color="#4B91F7",
456        )
457
458        # Add labels to the non-zero frequency histogram bars
459        for bin_value, patch in zip(n, self.patches):
460            x = patch.get_x() + patch.get_width() + 0.1
461            y = patch.get_y() + patch.get_height() / 2
462            self._ax_hist.annotate(
463                f"{bin_value}",
464                xy=(x, y),
465                xycoords="data",
466                xytext=(3, 0),
467                textcoords="offset points",
468                ha="left",
469                va="center",
470            )
471
472            # Calculate the midpoints between bin edges
473            bin_midpoints = (self.bins[:-1] + self.bins[1:]) / 2
474            # Set y-ticks at the bin midpoints
475            self._ax_hist.set_yticks(bin_midpoints)
476            # Set custom tick labels
477            custom_labels = [f"{bin_start} - {bin_end}" for bin_start, bin_end in zip(self.bins[:-1], self.bins[1:])]
478            self._ax_hist.set_yticklabels(custom_labels)

480    def on_canvas_click(self, event):
481        x_clicked, y_clicked = event.xdata, event.ydata
482        # Clear the previous clicked point
483        if self.clicked_point_scatter is not None:
484            try:
485                self.clicked_point_scatter.remove()
486            except Exception as e:
487                constants.PYSSA_LOGGER.warning(f"Something went wrong: {e}")
488
489        # Find the nearest point on the line
490        distance_data = self.protein_pair_for_analysis.distance_analysis.analysis_results.distance_data[pyssa_keys.ARRAY_DISTANCE_DISTANCES]
491        x_line = range(len(distance_data))
492        y_line = distance_data
493        distances = np.sqrt((x_line - x_clicked) ** 2 + (y_line - y_clicked) ** 2)
494        index_nearest = np.argmin(distances)
495        x_nearest, y_nearest = x_line[index_nearest], y_line[index_nearest]
496        self.clicked_point_scatter = self._ax_plot.scatter(x_nearest, y_nearest, color='red', marker='o', label='Clicked Point')
497        self.plot_widget.canvas.draw()
498        self.active_row_information = self.search_point_by_id(x_nearest)
499        if self.active_row_information is not None:
500            tmp_prot_1_name = self.protein_pair_for_analysis.protein_1.get_molecule_object()
501            tmp_prot_2_name = self.protein_pair_for_analysis.protein_2.get_molecule_object()
502            msg: str = (f"Status: Residue pair no. = {x_nearest}, Distance (Å) = {y_nearest} Between {tmp_prot_1_name} "
503                        f"Chain {self.active_row_information[1]} Position {self.active_row_information[2]} Residue {self.active_row_information[3]} and "
504                        f"{tmp_prot_2_name} Chain {self.active_row_information[4]} Position {self.active_row_information[5]} Residue {self.active_row_information[6]}")
505            self.lbl_status.setText(msg)
506            if self._sync_with_pymol_flag:
507                self.highlight_residue_in_pymol(x_nearest, tmp_prot_1_name)
508        else:
509            self.lbl_status.setText(f"Status: Residue pair no. = {x_nearest}, Distance (Å) = {y_nearest}")
510        # Change the selection color
511        self.change_selection_color(QtGui.QColor(75, 145, 247, 200))
512        self.highlight_histogram_bar(y_nearest)
