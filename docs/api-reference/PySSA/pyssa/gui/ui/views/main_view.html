232    def build_proteins_table(self):
233        self.cb_chain_color = QtWidgets.QComboBox()
234        self.cb_chain_representation = QtWidgets.QComboBox()
235
236        self.ui.proteins_table_widget.verticalHeader().setVisible(False)
237        self.ui.proteins_table_widget.setColumnCount(2)
238        self.ui.proteins_table_widget.setHorizontalHeaderLabels(["Name", "Value"])
239        gui_utils.fill_combo_box(self.cb_chain_color, constants.PYMOL_COLORS)
240        self.cb_chain_color.adjustSize()
241        gui_utils.fill_combo_box(self.cb_chain_representation, constants.PYMOL_REPRESENTATIONS)
242        self.cb_chain_representation.adjustSize()
