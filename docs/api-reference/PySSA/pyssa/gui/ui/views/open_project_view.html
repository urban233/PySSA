50    def _initialize_ui(self) -> None:
51        """Initialize the UI elements."""
52        self.ui.lbl_open_status_search.setText("")
53        self.ui.projects_list_view.setEditTriggers(QtWidgets.QAbstractItemView.NoEditTriggers)
54        self.ui.btn_help.setIcon(QtGui.QIcon(":/icons/help_w200.svg"))

56        self.ui.btn_help.setText("")
57        styles.set_stylesheet(self)
58        styles.color_bottom_frame_button(self.ui.btn_open_project)
59        self.setWindowIcon(QtGui.QIcon(constants.PLUGIN_LOGO_FILEPATH))
60        self.setWindowTitle("Open Project")
61        self.setWindowFlags(self.windowFlags() ^ QtCore.Qt.WindowContextHelpButtonHint)
62
63    def closeEvent(self, event):
